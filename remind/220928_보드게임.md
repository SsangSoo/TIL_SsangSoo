package com.codestates.coplit; 
import java.util.*;

public class Solution { 
	public Integer boardGame(int[][] board, String operation) {
    // TODO:
    // 사람의 언어 로직
    // 움직임마다, 0과 1을 보고 카운팅한 후, 점수 내기
    // 만약 보드판 벗어나면, 나가리..

    // 컴퓨팅 언어 로직.
    // 2차원배열 board[0][0]을 시작점으로,
    // 문자열 operation에 맞게 움직이고, 
    // index를 확인 (보드 안팎으로..)
    // 0인지, 1인지 판별해서, 
    // counting에 더함..사실 0이 나오더라도, 혹은 1이 나오더라도, 더하면 됨.

    String[] direction = operation.split(""); // operation을 빈문자열로 나눠서, String배열로 저장.

    int n = board.length;

    int sum = 0; // 점수계산

    int x = 0;  // x축 시작점 - 열[][*]
    int y = 0;  // y축 시작점 - 행[*][]

    for(int i=0; i<direction.length; i++) {   // direction을 돌면서,
      
      switch(direction[i]) {
        case "U" :    // U 위로 가야함.
          y -= 1;     //  행을 위로.. 빼기
          break;      
        case "D" :    // D 밑으로 가야함.
          y += 1;     //  행을 밑으로 더하기
          break;
        case "R" :    // R 오른쪽으로 가야함.
         x += 1;      //  열을 오른쪽으로 이동하기 위해 더하기.
         break;
        case "L" :    // L 왼쪽으로 가야함.
         x -= 1;      //  열을 왼쪽으로 이동하기 위해 빼기.
         break;
      }
      if( x < 0 || x >= n || y < 0 || y >= n) { // switch를 통해서, 칸 이동후, 배열 범위 벗어낫는지 확인. 벗어났으면, null 반환
        return null;
      } else { sum += board[y][x];  }
	 } // for 
  return sum;
  } // method
} // class
